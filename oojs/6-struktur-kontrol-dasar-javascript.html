        6. [OOJS] Struktur Kontrol Dasar

        Photo by Irvan Smith on Unsplash

        Struktur kontrol dalam pemrograman terdapat dua hal penting yaitu percabangan dan pengulangan

        # Percabangan

        Percabangan digunakan apabila kita dihadapkan pada 2 kondisi atau lebih dan harus memilih alur diantara keduanya menggunakan percabangan. Terdapat setidaknya 3 jenis percabangan

        1. If
        2. If-Else
        3. If-Else if-else

        Kalian telah sering menggunkanan if-else di materi-materi selanjutnya. Struktur dasar dari percabangan adalah If(Statement) { expresssion }. Kondisi dalam expression akan di eksekusi jika statement menghasilkan nilai true.

        Selain itu, kita juga telah berkenalan dengan operator ternary, yang mana dapat menggantikan fitur if-ese. Lihat materi 3.

        Pada pemrograman tingkat lanjut, dikenal istilah Branchless Programming dimana pemrograman dengan sebisa mungkin menghindari if statement. Karena penggunaan If dianggap memperlambat eksekusi program, bahkan hanya untuk 0,1ms.

        Namun untuk tujuan belajar, kita boleh menggunakan apapun, dan perlahan-lahan berorientasi menuju Branchless programming. Atau hal minimal yang bisa kita lakukan jika menemui percabangan adalah dengan menghapus kata else. Jadi hanya ada satu if dengan satu eksekusi program.

        # if

        // Contoh penggunaan if/if-else/if-else-if
class UtsPbo {
	constructor(nama, nilaiUtsPbo) {
		this.nama = nama;
		this.nilai = nilaiUtsPbo;
	}

	// Hanya ada satu if, jika kondisi tidak terpenuhi maka
	// Kode di dalamnya akan diabaikan
	cek1() {
		if (this.nilai > 85) {
			// Jika kondisi terpenuhi
			console.log(`${this.nama}, Kamu mendapat nilai A`);
		}
	}

	cek2() {
		if (this.nilai > 85) {
			// Jika terpenuhi
			console.log(`${this.nama}, Kamu mendapat nilai A`);
		} else {
			// Jika tidak terpenuhi
			console.log(`${this.nama}, Kamu tidak mendapat nilai A`);
		}
	}

	cek3() {
		if (this.nilai > 85) {
			// Jika terpenuhi
			console.log(`${this.nama}, Kamu mendapat nilai A`);

			// Jika tidak terpenuhi lakukan pengecekan kembali
		} else if (this.nilai > 75) {
			// Jika pengecekan kedua terpenuhi
			console.log(`${this.nama}, Kamu mendapat nilai B`);
		} else {
			// jika tidak terpenuhi
			console.log(`${this.nama}, Kamu tidak mendapat nilai A maupun B`);
		}
	}
}

const Ani = new UtsPbo("Ani", 70);
Ani.cek1();
Ani.cek2();
Ani.cek3();

        If-Else if-Else

        Bisa memiliki banyak else if. Misal If (){}else if(){}else if(){}else if(){}else{}.

        Seperti yang kita bahas di materi sebelumnya If juga bisa diubah menjadi ternary operator, namun hanya untuk If-else. Untuk if-else if-else, bisa digunakan nested ternary dan juga nested if atau percabangan bersarang dimana If berada di dalam if lagi.

        # switch

        Selain menggunakan if, kalian bisa mengubah program if yang terlalu panjang menggunakan switch.

        class UtsPbo {
	constructor(nama, nilaiUtsPbo) {
		this.nama = nama;
		this.nilai = nilaiUtsPbo;
	}

	cek() {
		let nilai = this.nilai;

		// Perhatikan dan analisis kode switch dengan teliti
		// Switch() hanya menerima ekspresi, kemudian ekspresi
		// di evaluasi menggunakn case :
		// dan break untuk keluar dari switch
		// default sama saja dengan else

		switch (nilai) {
			case nilai > 85:
				console.log(`${this.nama}, Kamu mendapat nilai A`);
				break;
			case nilai > 75:
				console.log(`${this.nama}, Kamu mendapat nilai B`);
				break;
			case nilai > 60:
				console.log(`${this.nama}, Kamu mendapat nilai C`);
				break;
			default:
				console.log(`${this.nama}, Kamu mendapat nilai D`);
				break;
		}
	}
}

const Ani = new UtsPbo("Ani", 70);
Ani.cek();

        # Perulangan

        Selanjutnya struktur kontrol yang penting untuk dipelajari dalam pemrograman. Bayangkan apabila kita punya database nama mahasiswa sebanyak 100 orang dan kita perlu menampilkannya ke layar nama-nama tersebut, pasti sangat merepotkan. Dengan perulangan kita bisa menjawab permasalahan tersebut.

        Perulangan dasar terbagi menjadi 3

        1. for
        2. do while
        3. while

        Perlu diingat bahwa perulangan memerlukan kondisi yang harus dipenuhi untuk melakukan perulangan dan kondisi yang tidak dipenuhi untuk mengakhiri perulangan. Hati-hati agar tidak terjebak dalam infinity loop.

        # for

        Perulangan for memiliki struktur sintaks seperti berikut

        # Perulangan for naik

        for(let i = 0; i <= ...; i++){
    // kode
}
        1
        2
        3

        # Perulangan for turun

        for(let i = ...; i >= 0; i--){
    // kode
}
        1
        2
        3
        • let i = mendeklarasikan variable i sebagai nilai awal
        • i >= atau i <= = menentukan batas dari perulangan
        • i-- atau i++ = menambah atau mengurangi i setiap kali 1 perulangan selesai

        # Contoh program

        class Perulangan {
	main() {
		for (let a = 1; a <= 10; a++) {
			console.log(`Saya berjanji akan mengerjakan semua challenge`);
		}
	}
}

const coba = new Perulangan();

coba.main();

        # Do While dan While

        Perulangan do-while dan while kurang lebih sama, perbedaan yang paling mendasar adalah, do-while akan melakukan eksekusi block program sekali, lalu melakukan evaluasi kondisi. Sementara while akan langsung melakukan evaluasi untuk menentukan apakah akan melakukan perulangan atau tidak.

        Struktur do while dan while

        do {
	// lakukan sekali
} while (kondisi);

//===============

while (kondisi) {
	// lakukan

        # Contoh program

        class Perulangan {
	contohDoWhile() {
		let itachi = 20;
		let sasuke = 10;

		do {
			console.log("Sasuke lebih hebat dari Itachi");
		} while (sasuke > itachi);

		// Meski itachi lebih besar nilainya dari sasuke
		// Namun "sasuke lebih hebat dari itachi" akan di tampilkan sekali
	}

	contohWhile() {
		let itachi = 20;
		let sasuke = 10;

		while (sasuke > itachi) {
			console.log("Sasuke lebih hebat dari Itachi");
		}

		// Sasuke lebih hebat dari itachi tidak akan ditampilkan ke layar
	}

	contohWhileLangsung() {
		let escanor = 10;
		let batas = 20;

		console.log("=======================================");
		console.log(`Kekuatan escanor: ${escanor}`);
		console.log("=======================================");

		// selama nilai escanor kurang dari nilai batas

		while (escanor < batas) {
			console.log(`Kekuatan escanor bertambah menjadi ${escanor}`);

			escanor++;

			// Tambah kekuatan escanor setiap perulangan
			// sampai escanor melebihi nilai batas, maka perulangan berhenti
		}

		console.log(`Kekuatan Escanor sekarang: ${escanor}`);
	}
}

const coba = new Perulangan();

coba.contohDoWhile();
coba.contohWhile();
coba.contohWhileLangsung();

        # Challenge

        Jalankan program di bawah ini, analisis dan ubahlah sehingga menjadi program milik kalian dengan menggunakan perulangan dan percabangan. Ubah tema, jumlah pilihan, jumlah method, jumlah property menjadi satu kesatuan program yang utuh.

        GUNAKAN NODE.JS

        Karena program di bawah ini menggunakan readline-sync, gunakan teks editor dan juga node. Pastikan sudah menginstall npm install readline-sync seperti di materi ke-5.

        // PROGRAM RESTORAN
let beri = require("readline-sync");

class Restoran {
	nama;
	menu;
	totalBayar;

	constructor(nama) {
		this.nama = nama;
		this.totalBayar = 0;

		this.menu = [
			{
				makanan: "ayam bakar",
				harga: 10_000,
			},
			{
				makanan: "nasi bakar",
				harga: 12_000,
			},
		];
	}

	welcome() {
		console.log(`Selamat datang di restoran ${this.nama}`);
		this.tampilkanMenu();
	}

	tampilkanMenu() {
		console.log("=======================================");
		for (let i = 0; i <= this.menu.length - 1; i++) {
			console.log(`${i}. ${this.menu[i].makanan} - Rp. ${this.menu[i].harga}`);
		}

		this.berikanPilihan();
	}

	berikanPilihan() {
		console.log("=======================================");
		let pilihan = beri.question(`Apa yang ingin anda pesan? `);
		console.log("=======================================");

		if (this.menu[pilihan]) {
			this.totalBayar += this.menu[pilihan].harga;
			console.log(`Terima kasih telah memesan ${this.menu[pilihan].makanan}`);
			console.log(`Harga pesanan adalah Rp. ${this.menu[pilihan].harga}`);
			console.log("=======================================");

			let pesanLagi = beri.question(
				`Apakah anda ingin memesan makanan lagi? (y/n) `
			);

			if (pesanLagi == "y") return this.tampilkanMenu();
			return this.total();
		}

		console.log("Mohon berikan pesanan yang masuk akal!");
		this.berikanPilihan();
	}

	total() {
		console.log(`Terima kasih telah makan di restoran ${this.nama}`);
		console.log(`Silahkan membayar Rp. ${this.totalBayar}`);
		console.log("=======================================");
		console.log(`Terima kasih`);
		return;
	}
}

const alelResto = new Restoran("Awal");
alelResto.welcome();
