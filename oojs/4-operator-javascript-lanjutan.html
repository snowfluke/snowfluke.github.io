              # Operasi aritmatika antar tipe data

              Operasi antar tipe data merupakan operasi antara 2 buah operand atau lebih yang memiliki tipe data yang berbeda. Berikut adalah behavior atau perilaku operasi aritmatika antar tipe data berbeda yang wajib kalian ketahui dalam javascript.

              PENTING

              Simbol + memiliki dua kegunaan yaitu sebagai operator artimatika untuk penjumlahan dan sebagai Konkatinasi atau operator penggabungan string.

              # Operator aritmatika number

              Semua operator aritmatika selain + merupakan operator milik tipe data number. Jika tipe data lain seperti string, object, array di kenai operasi -, *, / dan % maka akan menghasilkan output NaN atau Not a Number

              // Operasi aritmatika number
class Test{
    main() {

        let angka1 = 7
        let angka2 = 3
        let string1 = 'cortana'
        let objek1 = {nama: 'sasuke'}
        let array1 = ['mobil', 'motor']

        console.log('Perkalian angka*angka: ', angka1*angka2 )
        // Operasi simbol -, *, / dan % dibawah akan menghasilkan NaN
        // Karena operator tersebut khusus untuk angka
        console.log('Perkalian angka*string: ', angka1*string1 )
        console.log('Perkalian angka*array: ', angka1*array1 )
        console.log('Perkalian angka*object: ', angka1*objek1 )

    }
}

              # Perilaku boolean

              Apabila tipe data boolean dikenai operasi aritmatika, maka secara otomatis mengkonversi nilai jika true maka menjadi 1 dan jika false akan menjadi 0.

              # Operator konkatinasi

              Operator konkatinasi adalah operator yang bisa digunakan di seluruh tipe data dengan tujuan untuk menggabungkan string. Semua tipe data apabila diberikan operasi + dengan sebuah string, maka akan berubah menjadi string.

              // Operator konkatinasi
class Test{
    main() {

        let string1 = 'cortana'
        let angka1 = 7
        let objek1 = {nama: 'sasuke'}
        let array1 = ['mobil', 'motor']
        let boolean1 = true

        console.log('Konkatinasi string+string: ', string1+string1 )

        // Operasi simbol + pada semua operator akan menjadi sebuah string
        console.log('Konkatinasi string+angka: ', string1+angka1 )

        // untuk array akan dipisahkan dengan ',' tiap elemennya
        console.log('Konkatinasi string+array: ', string1+array1 )

        // Untuk boolean, yaitu stringnya (bukan angka 1 atau 0)
        console.log('Konkatinasi string+boolean: ', string1+boolean1 )
        
        // Untuk object, akan di ubah menjadi string [object Object]
        console.log('Konkatinasi string+object: ', string1+objek1 )

    }
}

              # Konversi string dan number

              Kalian bisa mengubah tipe data string menjadi number atau sebaliknya dengan menggunakan fungsi .parseInt() dan .toString() atau dengan menggunakan operator + dan - dengan cepat.

              # Konversi menggunakan fungsi

              class Test{
    main() {

        let a = '456'
        let b = 123

        console.log('Tipe data a: ', a)
        console.log('Tipe data b: ', b)

        // Mengubah a dari string ke number
        let c = parseInt(456)

        // Mengubah b dari number ke string
        let d = b.toString()

        console.log('Setelah diubah')
        console.log('Tipe data a: ', c)
        console.log('Tipe data b: ', d)

    }
}

              # Konversi menggunakan operasi

              class Test{
    main() {

        let a = '456'
        let b = 123

        console.log('Tipe data a: ', a)
        console.log('Tipe data b: ', b)

        // Mengubah a dari string ke number
        // Ingat sifat diatas, -, /, *, % hanya milik number
        let c = -a

        // Mengubah b dari number ke string
        // Ingat sifat diatas, semua yang di + string akan menjadi string
        let d = b+''

        console.log('Setelah diubah')
        console.log('Tipe data a: ', c)
        console.log('Tipe data b: ', d)

    }
}

              # Menggunakan variable di dalam string

              Jika kalian ingin membuat string dinamis yang akan berubah sesuai inputan user, misalkan Halo nama, selamat datang. Maka diperlukan sebuah variable dan variable tersebut harus digabungkan.

              Terdapat dua cara untuk menggabungkan variable dengan string.

              1. Menggunakan konkatinasi
              class Salam{
    halo(){
        // Penggabungan dengan menggunakan konkatinasi (+)
        let nama = 'Tsukishima'
        let tinggi = 190

        let salam = 'Halo '+ nama +', tinggimu adalah '+tinggi+'cm'
        
        console.log(salam)
    }
}

const tsukishima = new Salam()
tsukishima.halo()

              class Karyawan {
    nama,
    gaji
    lamaKerja
    tingkahLaku

    constructor(nama, lamaKerja, tingkahLaku){
        this.nama = -nama,
        this.lamaKerja = lamaKerja,
        this.tingkahLaku = tingkahLaku
    }

    gajian(){
        let gajiPokok = 3_000_000
        let rumusGaji = gajiPokok + (this.lamaKerja * 100000)
        
        let gajiAkhir = rumusGaji

        if(tingkahLaku) {
            this.tingkahLaku = 'baik'
            gajiAkhir += 250000
        }else{
            this.tingkahLaku = 'buruk'
            gajiAkhir += 100000
        }

        let selamat = `Selamat! ${nama}, sekarang adalah hari gajianmu! XD :D`
        
        console.log(selamat)
        console.log('====================================')
        console.log(`Karena kamu telah bekerja selama ${this.lamaKerja} bulan`)
        console.log(`Dan berperilaku ${this.tingkahLaku}`)
        console.log("Maka gajimu adalah Rp. "+ gajiAkhir +' rupiah')
        console.log('====================================')
        console.log(' ')
    }

}
